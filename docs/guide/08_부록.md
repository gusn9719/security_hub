## 실행 명령어

```powershell
# 백엔드
cd C:\dev\security_hub\backend
venv\Scripts\Activate.ps1
uvicorn main:app --reload --port 8000

# NF 벤치마크 (주의: 그대로 돌리면 자체 레이트리밋 충돌로 대부분 FAIL —

python backend/tests/run_qa_benchmark.py

# 시스템 레벨 3분류 정확도 (헤드라인 — 실제 analyze() 파이프라인 호출)
cd C:\dev\security_hub\backend
python tests/run_performance_eval.py    # 기존셋 508건, 정확도 100%
python tests/run_holdout_eval.py        # 독립 holdout 250건, SRS 목표 PASS
python tests/run_combined_eval.py       # (주의: holdout 0건 로딩 버그 있음)

# 휴리스틱 컴포넌트 단위 분석 + 투표 톤완화/승급 실증 (서버 불필요)
cd C:\dev\security_hub
python backend/tests/run_heuristic_eval.py
python backend/tests/run_vote_escalation_test.py
python backend/tests/run_safe_tone_down_test.py

# 테스트 (휴리스틱 25종 시그널 단위테스트 포함)
cd backend
python -m pytest tests/ -v
python -m pytest tests/test_heuristic_scorer.py -v

# Flutter
cd frontend
flutter run
```
